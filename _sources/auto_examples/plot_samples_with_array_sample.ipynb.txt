        "from __future__ import annotations\n\nimport matplotlib.pyplot as plt\nimport numpy as np\n\nfrom probly.representation.sampling.sample import ArraySample\n\n# Imagine these are 3 stochastic forward passes for 2 instances and 4 classes.\n# Shape per pass: (instances, classes)\npass_1 = np.array([[0.1, 0.2, 0.6, 0.1], [0.7, 0.1, 0.1, 0.1]])\npass_2 = np.array([[0.2, 0.2, 0.5, 0.1], [0.6, 0.2, 0.1, 0.1]])\npass_3 = np.array([[0.15, 0.25, 0.5, 0.1], [0.65, 0.15, 0.1, 0.1]])\n\ndata_stack = np.array([pass_1, pass_2, pass_3])\n\n\nsample = ArraySample(data_stack, sample_axis=0)\n\nmean = np.mean(data_stack, axis=0)\nstd = np.std(data_stack, axis=0, ddof=0)\n\nprint(\"mean shape:\", mean.shape)\nprint(\"std shape:\", std.shape)\nprint(\"mean[0]:\", mean[0])\nprint(\"std[0]:\", std[0])\n\n# Plot mean \u00b1 std for instance 0\nclasses = np.arange(mean.shape[1])\nplt.figure(figsize=(6, 3))\nplt.errorbar(classes, mean[0], yerr=std[0], fmt=\"o\", capsize=4)\nplt.xticks(classes)\nplt.ylim(0, 1)\nplt.xlabel(\"Class index\")\nplt.ylabel(\"Probability\")\nplt.title(\"Instance 0: mean \u00b1 std across stochastic passes\")\nplt.tight_layout()"
